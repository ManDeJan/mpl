template <typename F, typename C = identity>
using all = find_if<detail::not_<F>, always<bool_<false>, C>, always<bool_<true>, C>>;

tests if a predicate holds for all elements in a pack

Effects: resolves to true_ if all elements in the input pack fulfill the provided predicate, otherwise false_. \requires Type F shall be a continuation predicate and C shall be any continuation. example: call<all<same_as<void>>,void,void,void> resolves to true_.
